// create topic
docker exec broker-kafka \
kafka-topics --bootstrap-server broker-kafka:9092 \
             --create \
             --replication-factor 1 \
             --topic quickstart


// show topic list
docker exec broker-kafka \
kafka-topics --bootstrap-server broker-kafka:9092 \
             --list

// alter add partition
docker exec broker-kafka \
kafka-topics \
--alter \
--bootstrap-server broker-kafka:9092 \
--topic arkadia-analytics \
--partitions 2
